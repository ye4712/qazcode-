📚 Синтаксис Мысалдары

1️⃣ Бірінші Бағдарлама

басып_шығару("Сәлем, Әлем!");

2️⃣ Айнымалыларды Қолдану

х = 10;
у = 20;
қосынды = х + у;
басып_шығару(қосынды);

3️⃣ Шартты Операторлар

х = 15;
егер х > 10:
    басып_шығару("х үлкен!");

4️⃣ Математикалық Операциялар

а = 20;
б = 4;
басып_шығару(а + б);  # 24
басып_шығару(а - б);  # 16
басып_шығару(а * б);  # 80
басып_шығару(а / б);  # 5
